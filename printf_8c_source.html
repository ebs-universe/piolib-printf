51static int printchar(void **target, const char ttype, int c)
52{
53 extern int putchar(int c);
54 char **str;
55 bytebuf *buf;
56 if (ttype == PRINT_TTYPE_STRING){
57 str = (char**)target;
58 if (str) {
59 **str = c;
60 ++(*str);
61 }
62 }
63 else if (ttype == PRINT_TTYPE_BYTEBUF){
64 buf = *((bytebuf**)target);
65 return bytebuf_cPushByte(buf, c, BYTEBUF_TOKEN_PRINT);
66 }
67 return putchar(c);
68}

73static int prints(void **out, const char ttype, const char *string, int width, int pad)
74{
75 register int pc = 0, padchar = ' ';
76
77 if (width > 0) {
78 register int len = 0;
79 register const char *ptr;
80 for (ptr = string; *ptr; ++ptr) ++len;
81 if (len >= width) width = 0;
82 else width -= len;
83 if (pad & PAD_ZERO) padchar = '0';
84 }
85 if (!(pad & PAD_RIGHT)) {
86 for ( ; width > 0; --width) {
87 pc += printchar (out, ttype, padchar);
88 }
89 }
90 for ( ; *string ; ++string) {
91 pc += printchar (out, ttype, *string);
92
93 }
94 for ( ; width > 0; --width) {
95 pc += printchar (out, ttype, padchar);
96 }
97
98 return pc;
99}

101static int printl(void **out, const char ttype, long i, int b, int sg, int width, int pad, int letbase)
102{
103 char print_buf[PRINT_BUF_LEN];
104 register char *s;
105 register int t, neg = 0, pc = 0;
106 register unsigned long u = i;
107
108 if (i == 0) {
109 print_buf[0] = '0';
110 print_buf[1] = '\0';
111 return prints (out, ttype, print_buf, width, pad);
112 }
113
114 if (sg && b == 10 && i < 0) {
115 neg = 1;
116 u = -i;
117 }
118
119 s = print_buf + PRINT_BUF_LEN-1;
120 *s = '\0';
121
122 while (u) {
123 t = u % b;
124 if( t >= 10 )
125 t += letbase - '0' - 10;
126 *--s = t + '0';
127 u /= b;
128 }
129
130 if (neg) {
131 if( width && (pad & PAD_ZERO) ) {
132 pc += printchar (out, ttype, '-');
133 --width;
134 }
135 else {
136 *--s = '-';
137 }
138 }
139
140 return pc + prints (out, ttype, s, width, pad);
141}

193static int printfloat(void **out, const char ttype, double f, int width, int pad, int precision) {
194 // See https://stackoverflow.com/a/77372106/1934174
195 register int neg = 0, pc = 0, p = 0;
196 register uint32_t digits, fraction, exponent;
197 float mantissa;
198
199 if (f < 0) {
200 f = -f;
201 neg = 1;
202 }
203
204 if (f == 0) {
205 p = 12;
206 } else {
207 for (p = 0; p < sizeof(pwr10) / sizeof(pwr10[0]) - 1; p++) {
208 if (f >= pwr10[p]) break;
209 }
210 }
211
212 exponent = 36 - 3 * p;
213 mantissa = f / pwr10[p];
214
215 mantissa += 0.00005; // 4 digit precision
216 digits = mantissa;
217 fraction = (mantissa - digits) * 10000.0;
218
219 if (neg){
220 pc += printchar(out, ttype, '-');
221 }
222 pc += printi(out, ttype, digits, 10, 1, 0, 0, 'a');
223 pc += printchar(out, ttype, '.');
224 pc += printi(out, ttype, fraction, 10, 1, 4, PAD_ZERO, 'a');
225 if (exponent){
226 pc += printchar(out, ttype, 'e');
227 pc += printi(out, ttype, exponent, 10, 1, 0, 0, 'a');
228 }
229 return pc;
230}

235int print(void **out, const char ttype, const char *format, va_list args )
236{
237 register int width, pad;
238 register int pc = 0;
239 char scr[2], lng=0;
240 char** outp;
241
242 for (; *format != 0; ++format) {
243 if (*format == '%') {
244 ++format;
245 width = pad = 0;
246 #if PRINT_SUPPORT_FLOAT
247 int precision = -1;
248 #endif
249 if (*format == '\0') break;
250 if (*format == '%') goto out;
251 if (*format == '-') {
252 ++format;
253 pad = PAD_RIGHT;
254 }
255 while (*format == '0') {
256 ++format;
257 pad |= PAD_ZERO;
258 }
259 for ( ; *format >= '0' && *format <= '9'; ++format) {
260 width *= 10;
261 width += *format - '0';
262 }
263 #if PRINT_SUPPORT_FLOAT
264 if (*format == '.') {
265 ++format;
266 precision = 0;
267 for (; *format >= '0' && *format <= '9'; ++format) {
268 precision *= 10;
269 precision += *format - '0';
270 }
271 }
272 #endif
273 if( *format == 'l' ) {
274 lng=1;
275 ++format;
276 }
277 #if PRINT_SUPPORT_FLOAT
278 if (*format == 'f') {
279 double fval = va_arg(args, double);
280 if (precision == -1) precision = PRINT_DEFAULT_FLOAT_PRECISION;
281 pc += printfloat(out, ttype, fval, width, pad, precision);
282 continue;
283 }
284 #endif
285 if( *format == 's' ) {
286 register char *s = (char *)va_arg( args, int );
287 pc += prints (out, ttype, s?s:"(null)", width, pad);
288 continue;
289 }
290 if( *format == 'd' ) {
291 if(!lng)
292 pc += printi (out, ttype, va_arg( args, int ), 10, 1, width, pad, 'a');
293 else
294 pc += printl (out, ttype, va_arg( args, long ), 10, 1, width, pad, 'a');
295 continue;
296 }
297 if( *format == 'x' ) {
298 pc += printi (out, ttype, va_arg( args, int ), 16, 0, width, pad, 'a');
299 continue;
300 }
301 if( *format == 'X' ) {
302 pc += printi (out, ttype, va_arg( args, int ), 16, 0, width, pad, 'A');
303 continue;
304 }
305 if( *format == 'u' ) {
306 pc += printi (out, ttype, va_arg( args, int ), 10, 0, width, pad, 'a');
307 continue;
308 }
309 if( *format == 'c' ) {
310 /* char are converted to int then pushed on the stack */
311 scr[0] = (char)va_arg( args, int );
312 scr[1] = '\0';
313 pc += prints (out, ttype, scr, width, pad);
314 continue;
315 }
316 }
317 else {
318 out:
319 pc += printchar (out, ttype, *format);
320 }
321 }
322 if (ttype == PRINT_TTYPE_STRING){
323 outp = (char **) out;
324 **outp = '\0';
325 }
326 va_end( args );
327 return pc;
328}
